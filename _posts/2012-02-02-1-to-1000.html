#include <iostream>

using namespace std ;

int main ( int argc , const char * argv [ ] )
{
unsigned int number = 1 ;
unsigned int lastStep = 3 ;
unsigned int step = 0 ;

for ( number = 1 ; number <= 1000 ; number ++ )
{
cout << number ;

step ++ ;

if ( number < 1000 && ( ( lastStep == 3 && step == 5 ) || ( lastStep == 5 && step == 3 ) ) )
{
lastStep = step ;
step = 0 ;

cout << "*" ;
}
}

return 0 ;
}
