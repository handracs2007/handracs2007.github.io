#include <iostream>

using namespace std ;

int main ( int argc , const char * argv [ ] )
{
char letters [ ] = "AAAAAAAAAA" ;
int number = 0 ;

do
{
if ( number == 1000 )
{
number = 0 ;

for ( int i = ( int ) strlen ( letters ) - 1 ; i >= 0 ; i -- )
{
if ( letters [ i ] != 'Z' )
{
letters [ i ] ++ ;

break ;
}
}
}

cout << letters << ( ( number < 10 ) ? "00" : ( number < 100 ) ? "0" : "" ) << number << endl ;

number ++ ;

if ( strcmp ( letters , "ZZZZZZZZZZ" ) == 0 && number == 1000 )
break ;
} while ( true) ;

return 0 ;
}
